/// Controller do settings da APP

class SettingsProvider extends ChangeNotifier{
  bool lightTheme;

  setTheme(bool lightTheme) {
    try {
      this.lightTheme = lightTheme;
    } catch (e) {
      debugPrint(e.toString());
      this.lightTheme = true;
    }
    notifyListeners();
  }
}
