class DebugPrint {
  // Constante para determinar quando estamos em desenvolvimento ou producao
  static const bool debuggingMode = true;

  static void imprimir(String texto) {
    if (debuggingMode) {
      developer.log(texto.toLowerCase(), name: 'Debug:');
    }
  }

  static void error(String error) {
    if (debuggingMode) {
      String _error = "**********\n";
      _error += "##########\n";
      _error += "\n${error.toUpperCase()}\n";
      _error += "##########\n";
      _error += "**********";
      developer.log("erro", name: 'Error:', error: "****\n$_error\n****");
    }
  }
}
